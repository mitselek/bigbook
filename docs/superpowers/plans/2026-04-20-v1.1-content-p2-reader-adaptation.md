Replace the `<script>` block of `src/components/TocOverlay.svelte` (lines 1–104) with the content below. The template and styles (lines 105 onward) are unchanged.

```svelte
<script lang="ts">
  import { onMount } from 'svelte'
  import type { ChapterGroup, ChapterManifest } from '../lib/content/manifest'
  import { readerState } from '../lib/reader/store.svelte'

  interface Props {
    chapters: readonly ChapterManifest[]
    isOpen?: boolean
    onSelect?: (slug: string) => void
    onClose?: () => void
  }
  let { chapters, isOpen, onSelect, onClose }: Props = $props()

  let open = $derived(isOpen ?? readerState.tocOpen)

  function close() {
    readerState.tocOpen = false
    onClose?.()
  }

  function select(slug: string) {
    onSelect?.(slug)
    close()
    document.dispatchEvent(new CustomEvent('bigbook:force-load', { detail: { slug } }))
    const target = document.getElementById(`chapter-${slug}`)
    target?.scrollIntoView({ behavior: 'smooth', block: 'start' })
  }

  const GROUP_ORDER: readonly ChapterGroup[] = ['front-matter', 'chapters', 'stories', 'appendices']
  const GROUP_LABEL: Record<ChapterGroup, string> = {
    'front-matter': 'Front matter',
    chapters: 'Chapters',
    stories: 'Stories',
    appendices: 'Appendices',
  }

  function stripHash(title: string): string {
    return title.replace(/^#\s*/, '')
  }

  type Group = { label: string; items: ChapterManifest[] }

  const groups = $derived.by(() => {
    return GROUP_ORDER.map((g) => ({
      label: GROUP_LABEL[g],
      items: chapters.filter((ch) => ch.group === g),
    })).filter((group) => group.items.length > 0)
  })

  const entries = $derived(groups.flatMap((g) => g.items))

  let focusedIndex = $state(-1)
  let dialogEl = $state<HTMLElement | null>(null)

  $effect(() => {
    if (open && dialogEl) dialogEl.focus()
  })

  onMount(() => {
    const onToggle = () => {
      readerState.tocOpen = !readerState.tocOpen
    }
    document.addEventListener('bigbook:toggle-toc', onToggle)
    return () => document.removeEventListener('bigbook:toggle-toc', onToggle)
  })

  function handleKeydown(e: KeyboardEvent) {
    const len = entries.length
    if (e.key === 'ArrowDown') {
      focusedIndex = (focusedIndex + 1) % len
    } else if (e.key === 'ArrowUp') {
      focusedIndex = (focusedIndex - 1 + len) % len
    } else if (e.key === 'Enter' && focusedIndex >= 0) {
      const entry = entries[focusedIndex]
      if (entry) select(entry.slug)
    } else if (e.key === 'Escape') {
      close()
    } else if (e.key === 'Tab' && dialogEl) {
      e.preventDefault()
      const focusable = Array.from(
        dialogEl.querySelectorAll<HTMLElement>('button,[tabindex]'),
      ).filter((el) => el !== dialogEl)
      if (focusable.length === 0) return
      const current = focusable.indexOf(document.activeElement as HTMLElement)
      const next = e.shiftKey
        ? (current - 1 + focusable.length) % focusable.length
        : (current + 1) % focusable.length
      focusable[next]?.focus()
    }
  }
</script>
```

Key diff points vs current code:

- `FRONT_MATTER` / `APPENDICES` slug sets and `groupLabel(slug)` helper deleted.

- `groups` derivation rewritten: iterate `GROUP_ORDER`, filter `chapters` by `ch.group === g`, drop empty groups with `.filter((group) => group.items.length > 0)`.
